40void gpio_pin_init(GPIOx_Handle_t *GPIOx_Handle) {
41 uint8_t gpio_pin_number = GPIOx_Handle->GPIO_CONFIG.GPIO_PIN_NUMBER;
42 uint8_t gpio_mode = GPIOx_Handle->GPIO_CONFIG.GPIO_MODE;
43 uint8_t gpio_speed = GPIOx_Handle->GPIO_CONFIG.GPIO_SPEED;
44 uint8_t gpio_output_type = GPIOx_Handle->GPIO_CONFIG.GPIO_OP_TYPE;
45 uint8_t gpio_pu_pd = GPIOx_Handle->GPIO_CONFIG.GPIO_PU_PD;
46 uint8_t gpio_alt_func = GPIOx_Handle->GPIO_CONFIG.GPIO_ALT_FUNC;
47 /*1. Enable the respective GPIO peripheral through RCC */
48 if (GPIOx_Handle->pGPIOx == GPIOA) {
49 GPIOA_PCLK_EN();
50 } else if (GPIOx_Handle->pGPIOx == GPIOB) {
51 GPIOB_PCLK_EN();
52 } else if (GPIOx_Handle->pGPIOx == GPIOC) {
53 GPIOC_PCLK_EN();
54 } else if (GPIOx_Handle->pGPIOx == GPIOD) {
55 GPIOD_PCLK_EN();
56 } else if (GPIOx_Handle->pGPIOx == GPIOE) {
57 GPIOE_PCLK_EN();
58 } else if (GPIOx_Handle->pGPIOx == GPIOF) {
59 GPIOF_PCLK_EN();
60 } else if (GPIOx_Handle->pGPIOx == GPIOG) {
61 GPIOG_PCLK_EN();
62 } else if (GPIOx_Handle->pGPIOx == GPIOH) {
63 GPIOH_PCLK_EN();
64 } else if (GPIOx_Handle->pGPIOx == GPIOI) {
65 GPIOI_PCLK_EN();
66 }
67 /*2.Set the pin mode */
68 GPIOx_Handle->pGPIOx->MODER &= ~(0x03 << (2 * gpio_pin_number)); /* Clear the bits */
69 GPIOx_Handle->pGPIOx->MODER |= (gpio_mode << (2 * gpio_pin_number)); /* Set the desire bits */
70
71 /* 3. Set the GPIO Speed. @note: speed is applicable only for output and alt-function mode */
72 if (gpio_mode == GPIO_MODE_OUTPUT || gpio_mode == GPIO_MODE_ALT_FUNC) {
73 GPIOx_Handle->pGPIOx->OSPEEDR &= ~(0x03 << (2 * gpio_pin_number)); /* Clear the bits */
74 GPIOx_Handle->pGPIOx->OSPEEDR |= (gpio_speed << (2 * gpio_pin_number)); /* Set the desire bit */
75 }
76 /* 4. Set the output type (Push_pull/Open_Drain). @note This part is only applicable when mode is output or alternate Function */
77 if (gpio_mode == GPIO_MODE_OUTPUT || gpio_mode == GPIO_MODE_ALT_FUNC) {
78 GPIOx_Handle->pGPIOx->OTYPER &= ~(0x1 << gpio_pin_number); /* Clear the bits */
79 GPIOx_Handle->pGPIOx->OTYPER |= (gpio_output_type << gpio_pin_number); /* Set the desire bit */
80 }
81
82 /* 5. Set the pullup or pulldown config. @note this config is only applicable when mode is output or alternate function or input */
83 if (gpio_mode != GPIO_MODE_ANALOG) {
84 GPIOx_Handle->pGPIOx->PUPDR &= ~(0x03 << (2 * gpio_pin_number)); /* Clear the bits */
85 GPIOx_Handle->pGPIOx->PUPDR |= (gpio_pu_pd << (2 * gpio_pin_number)); /* Set the desire bit */
86 }
87
88 /* 6. Setting ALT FUNCTION config. @note this config is only applicable when mode is ALT FUNC MODE */
89 if (gpio_mode == GPIO_MODE_ALT_FUNC) {
90 if (gpio_pin_number <= 7) { /*We have to configure AFRL */
91 GPIOx_Handle->pGPIOx->AFRL &= ~(0xF << (4 * (gpio_pin_number % 8)));
92 GPIOx_Handle->pGPIOx->AFRL |= (gpio_alt_func
93 << (4 * (gpio_pin_number % 8)));
94 } else { /*We have to configure AFRH */
95 GPIOx_Handle->pGPIOx->AFRH &= ~(0xF << (4 * (gpio_pin_number % 8)));
96 GPIOx_Handle->pGPIOx->AFRH |= (gpio_alt_func
97 << (4 * (gpio_pin_number % 8)));
98 }
99 }

181void gpio_irq_config(GPIOx_RegDef_t *pGPIOx, uint8_t pin, uint8_t trigger,
182 uint8_t irq_priority) {
198 SYSCFG_PCLK_EN();
199
201 SYSCFG_RegDef_t *pSYSCFG = SYSCFG;
202
218 uint8_t sys_cfg_reg_number = pin / 4;
219
230 uint8_t gpio_port_map;
231
232 if (pGPIOx == GPIOA)
233 gpio_port_map = 0x00;
234 else if (pGPIOx == GPIOB)
235 gpio_port_map = 0x01;
236 else if (pGPIOx == GPIOC)
237 gpio_port_map = 0x02;
238 else if (pGPIOx == GPIOD)
239 gpio_port_map = 0x03;
240 else if (pGPIOx == GPIOE)
241 gpio_port_map = 0x04;
242 else if (pGPIOx == GPIOF)
243 gpio_port_map = 0x05;
244 else if (pGPIOx == GPIOG)
245 gpio_port_map = 0x06;
246 else if (pGPIOx == GPIOH)
247 gpio_port_map = 0x07;
248 else if (pGPIOx == GPIOI)
249 gpio_port_map = 0x08;
250 else
251 return;
252
263 uint8_t shift_amount = 4 * (pin % 4);
264
265 if (sys_cfg_reg_number == 0) {
267 pSYSCFG->EXTICR1 &= ~(0xF << shift_amount);
268 pSYSCFG->EXTICR1 |= (gpio_port_map << shift_amount);
269 } else if (sys_cfg_reg_number == 1) {
271 pSYSCFG->EXTICR2 &= ~(0xF << shift_amount);
272 pSYSCFG->EXTICR2 |= (gpio_port_map << shift_amount);
273 } else if (sys_cfg_reg_number == 2) {
275 pSYSCFG->EXTICR3 &= ~(0xF << shift_amount);
276 pSYSCFG->EXTICR3 |= (gpio_port_map << shift_amount);
277 } else if (sys_cfg_reg_number == 3) {
279 pSYSCFG->EXTICR4 &= ~(0xF << shift_amount);
280 pSYSCFG->EXTICR4 |= (gpio_port_map << shift_amount);
281 }
282
284 EXTI_RegDef_t *pEXTI = EXTI;
285 pEXTI->RTSR &= ~(1 << pin);
286 pEXTI->FTSR &= ~(1 << pin);
287 if (trigger == INTERRUPT_TRIGGER_TYPE_RISING) {
288 pEXTI->RTSR |= (1 << pin);
289 } else if (trigger == INTERRUPT_TRIGGER_TYPE_FALLING) {
290 pEXTI->FTSR |= (1 << pin);
291 } else if (trigger == INTERRUPT_TRIGGER_TYPE_BOTH) {
292 pEXTI->RTSR |= (1 << pin);
293 pEXTI->FTSR |= (1 << pin);
294 }
295
297 pEXTI->PR |= (1 << pin);
298
300 pEXTI->IMR |= (1 << pin);
301
303
304 uint8_t irq;
305
306 /* Step 1: Identify which IRQ vector corresponds to the EXTI line */
307 if (pin <= 4) {
308 /* EXTI0 IRQ6, EXTI1 IRQ7, ... EXTI4 IRQ10 */
309 irq = IRQ_NUM_EXTI0 + pin;
310 } else if (pin <= 9) {
311 /* EXTI59 all map to EXTI9_5 shared IRQ */
312 irq = IRQ_NUM_EXTI9_5;
313 } else {
314 /* EXTI1015 all map to EXTI15_10 shared IRQ */
315 irq = IRQ_NUM_EXTI15_10;
316 }
317
318 /* Step 2: Compute the NVIC IPR register and byte offset */
319 NVIC_RegDef_t *pNVIC = NVIC;
320 uint8_t ipr_index = irq / 4; // Each IPR holds 4 priority fields
321 uint8_t ipr_offset = (irq % 4) * 8; // Select correct byte
322 ipr_offset += 4; // Priority uses only upper 4 bits [7:4]
323
324 /*
325 * Step 3: Clear the old priority
326 * We clear a full byte (even though only 4 bits are used),
327 * because NVIC ignores the lower 4 bits anyway.
328 */
329 pNVIC->IPR[ipr_index] &= ~(0xFF << ipr_offset);
330
331 /*
332 * Step 4: Write new priority
333 * irq_priority is masked to 4 bits (015),
334 * then shifted into bits [7:4] of the correct byte.
335 */
336 pNVIC->IPR[ipr_index] |= ((irq_priority & 0x0F) << ipr_offset);
337}

344void gpio_irq_control(uint8_t pin, uint8_t en_di) {
363
364 NVIC_RegDef_t *pNVIC = NVIC;
365 EXTI_RegDef_t *pEXTI = EXTI;
366
367 if (en_di == ENABLE) {
368 // 1. Unmask the EXTI line in EXTI_IMR
369 pEXTI->IMR |= (1 << pin);
370 // 2. Enable the corresponding NVIC interrupt via NVIC_ISER
371 uint8_t irq;
372 if (pin <= 4) {
373 /* EXTI0 IRQ6, EXTI1 IRQ7, ... EXTI4 IRQ10 */
374 irq = IRQ_NUM_EXTI0 + pin;
375 pNVIC->ISER[0] |= (1 << irq);
376 } else if (pin <= 9) {
377 /* EXTI59 all map to EXTI9_5 shared IRQ */
378 irq = IRQ_NUM_EXTI9_5;
379 pNVIC->ISER[0] |= (1 << irq);
380 } else {
381 /* EXTI1015 all map to EXTI15_10 shared IRQ */
382 irq = IRQ_NUM_EXTI15_10;
383 pNVIC->ISER[1] |= (1 << (irq % 32));
384 }
385 // 3. Clear the Pending interrupt.
386 gpio_irq_clear(pin);
387 } else { //Disable interrupt
388 // 1. Mask the EXTI line in EXTI_IMR
389 pEXTI->IMR &= ~(1 << pin);
390 // 2. Disable the corresponding NVIC interrupt via NVIC_ISER
391 uint8_t irq;
392 if (pin <= 4) {
393 /* EXTI0 IRQ6, EXTI1 IRQ7, ... EXTI4 IRQ10 */
394 irq = IRQ_NUM_EXTI0 + pin;
395 pNVIC->ICER[0] |= (1 << irq);
396 } else if (pin <= 9) {
397 /* EXTI59 all map to EXTI9_5 shared IRQ */
398 irq = IRQ_NUM_EXTI9_5;
399 pNVIC->ICER[0] |= (1 << irq);
400 } else {
401 /* EXTI1015 all map to EXTI15_10 shared IRQ */
402 irq = IRQ_NUM_EXTI15_10;
403 pNVIC->ICER[1] |= (1 << (irq % 32));
404 }
405 // 3. Clear the Pending interrupt.
406 gpio_irq_clear(pin);
407 }
