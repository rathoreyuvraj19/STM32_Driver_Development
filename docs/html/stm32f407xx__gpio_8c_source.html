64 /* 2. Configure GPIO mode */
65 temp = (pGPIOHandle->GPIO_Config.GPIO_Mode << (2 * pGPIOHandle->GPIO_Config.GPIO_PinNumber));
66 pGPIOHandle->pGPIOx->MODER &= ~(0x3 << (2 * pGPIOHandle->GPIO_Config.GPIO_PinNumber)); // clear

71 pGPIOHandle->pGPIOx->OTYPER &= ~(0x1 << pGPIOHandle->GPIO_Config.GPIO_PinNumber); // clear
72 pGPIOHandle->pGPIOx->OTYPER |= temp;

83
