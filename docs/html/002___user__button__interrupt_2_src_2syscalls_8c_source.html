48void ITM_SendChar(uint8_t ch)
49{
50
51 //Enable TRCENA
52 DEMCR |= ( 1 << 24);
53
54 //enable stimulus port 0
55 ITM_TRACE_EN |= ( 1 << 0);
56
57 // read FIFO status in bit [0]:
58 while(!(ITM_STIMULUS_PORT0 & 1));
59
60 //Write to ITM stimulus port0
61 ITM_STIMULUS_PORT0 = ch;
62}
